create table employees_1(
emp_id int primary key,
empname varchar(20),
dept_id int,
salary decimal(10,2),
join_date DATE
);

create table departmentt(
dept_id int primary key,
dept_name varchar(20)
);
create table projectts(
proj_id int primary key,
proj_name varchar(20),
budget decimal(15,2)
);
CREATE TABLE Employees_Projects (
    emp_id INT NOT NULL,        -- Employee ID (assumed to be an integer)
    proj_id INT NOT NULL,       -- Project ID (assumed to be an integer)
    assigned_date DATE,        -- Date when the employee was assigned to the project
    role VARCHAR(30),         -- Role of the employee in the project (can store up to 100 characters)
    PRIMARY KEY (emp_id, proj_id),  -- Composite primary key (unique combination of empid and projid)
    FOREIGN KEY (emp_id) REFERENCES Employees_1(emp_id),  -- reference to the 'employees_1' table
    FOREIGN KEY (proj_id) REFERENCES Projectts(proj_id)  -- reference to the 'projectts' table
);
